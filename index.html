<script>

let currentScreen = "intro";

let selectedIndex = 0;

let chosenCountry = null;


function updateSelection() {

  const options = document.querySelectorAll("#"+currentScreen+" .option");

  options.forEach((opt,i)=> {

    opt.classList.toggle("selected", i===selectedIndex);

  });

}


function switchScreen(id) {

  document.querySelectorAll(".screen").forEach(s=>s.classList.remove("active"));

  document.getElementById(id).classList.add("active");

  currentScreen = id;

  selectedIndex = 0;

  updateSelection();

}


document.addEventListener("keydown", e => {

  const options = document.querySelectorAll("#"+currentScreen+" .option");


  if(e.key === "ArrowDown") {

    selectedIndex = (selectedIndex+1)%options.length;

    updateSelection();

  }

  if(e.key === "ArrowUp") {

    selectedIndex = (selectedIndex-1+options.length)%options.length;

    updateSelection();

  }


  if(e.key === "Enter") {

    if(currentScreen==="intro") {

      if(selectedIndex===0) switchScreen("country");

      else switchScreen("no");

    }

    else if(currentScreen==="no") {

      switchScreen("intro");

    }

    else if(currentScreen==="country") {

      chosenCountry = selectedIndex===0 ? "japan" : "vietnam";

      document.getElementById("bossText").innerText =

        chosenCountry==="japan"

        ? "Confirm Japan — for our fast forward september trip?"

        : "Confirm Vietnam — for a trip down memory lane?";

      switchScreen("boss");

    }

    else if(currentScreen==="boss") {

      if(selectedIndex===0) {

        if(chosenCountry==="japan") {

          switchScreen("japan");

          createPetals();

        } else {

          switchScreen("vietnam");

        }

      } else {

        switchScreen("intro");

      }

    }

  }

});


function createPetals() {

  for(let i=0;i<25;i++){

    let p=document.createElement("div");

    p.className="petal";

    p.innerHTML="🌸";

    p.style.left=Math.random()*100+"vw";

    p.style.animationDuration=(5+Math.random()*5)+"s";

    document.body.appendChild(p);

  }

}


updateSelection();
